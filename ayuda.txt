SELECT ID_PRODUCTO, ID, PRODUCTO, STOCK_INICIAL, ISNULL([INGRESO], 0) AS 'INGRESO', ISNULL([SALIDA], 0) AS 'SALIDA', ISNULL([MERMA], 0) AS 'MERMA', 
ISNULL([CORTESIA], 0) AS 'CORTESIA', STOCK, (STOCK_INICIAL + STOCK) STOCK_ACTUAL FROM
(
	SELECT M.ID_PRODUCTO, P.ID, P.PRODUCTO, P.STOCK_INICIAL, P.STOCK, T.TIPO, SUM(M.CANTIDAD) CANTIDAD FROM MOVIMIENTO M
	INNER JOIN PRODUCTO P ON M.ID_PRODUCTO = P.ID_PRODUCTO
	INNER JOIN TIPO T ON M.ID_TIPO = T.ID_TIPO
	WHERE CONVERT(DATE, M.FECHA_CREACION) BETWEEN '2021-06-10' AND '2021-06-10' AND P.ID_ALMACEN = '10'
	GROUP BY M.ID_PRODUCTO, P.ID, P.PRODUCTO, P.STOCK_INICIAL, P.STOCK, T.TIPO, M.CANTIDAD)
	AS T PIVOT (SUM(CANTIDAD) FOR T.TIPO IN([INGRESO],[SALIDA],[MERMA],[CORTESIA])
) AS PVT
